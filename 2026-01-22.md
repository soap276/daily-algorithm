## 1. 리스트에 특정 값이 있는지 확인하기 
``` python
def check_fruit(fruit_list, fruit):
    if fruit in fruit_list:   ## in 활용하는 법 익히기. 특정 값이 있는지 확인할 땐 in 사용!!
        return '있음'
    else: 
        return '없음'
    
fruits = ["사과", "딸기", "멜론", "포도", "오렌지"]

print(check_fruit(fruits, "사과")) # 있음
print(check_fruit(fruits, "망고")) # 없음
print(check_fruit(fruits, "오렌지")) # 있음
```

## 2. 리스트 합 구하기 (누적 합 개념 익히기)
``` python 
def sum_list(lst):
    sum = 0
    for list in lst:
        sum += list
    return sum ## return 줄바꿈 항상 주의!!!

print(sum_list([1,2,3,4,5])) # 15
print(sum_list([100,200,300])) # 600
print(sum_list([])) # 0
```

## 3. 리스트 길이 구하기
``` python
def len_list(lst):
    count = 0
    for list in lst:
        count += 1
    return count

print(len_list([1,2,3,4,5])) # 5
print(len_list([100,200,300])) # 3
print(len_list([])) # 0
```

## 4. 딕셔너리의 모든 key와 value 출력
``` python
def print_dict(dic):
    for i in range(len(dic)):
        print(f'key : {list(dic.keys())[i]} / value : {list(dic.values())[i]}')
        ## 딕셔너리 키값, 벨류값 빼는 형식 익히기!!!

print_dict({"번호" : 1, (1,3,4): "튜플은 키로 사용 가능", 1 : "숫자도 사용 가능"})
# key : 번호 / value : 1
# key : (1, 3, 4) / value : 튜플은 키로 사용 가능
# key : 1 / value : 숫자도 사용 가능
```

## 5. 문자열에서 모음 개수 세기
```python
def get_vowels_count(text):
    count = 0
    for tex in text:
        if tex in "aeiou":  ## 문자열 사용해서 문제 푸는 아이디어
            count += 1
    return count

print(get_vowels_count("hello")) # 2
print(get_vowels_count("world")) # 1
print(get_vowels_count("hello my name is minseok")) # 8 
```

## 6. 문자열에서 각 글자가 등장한 횟수 세기, 딕셔너리로 반환 ⭐
``` python 
def char_count(text):
    result = {} # 딕셔너리로 반환한다고 했으니 딕셔너리 함수 먼저 생성
    for tex in text: 
        if tex in result: # 이미 등장한 문자인지 확인. in 연산자는 딕셔너리에서 key 존재 여부 확인
            result[tex] += 1
        else:
            result[tex] = 1
    return result

print(char_count("가나다가나다")) # {'가': 2, '나': 2, '다': 2}
print(char_count("123123123")) # {'1': 3, '2': 3, '3': 3}
print(char_count("jazz is good")) # {'j': 1, 'a': 1, 'z': 2, ' ': 2, 'i': 1, 's': 1, 'g': 1, 'o': 2, 'd': 1}
```

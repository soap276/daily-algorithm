## 특정 거리의 도시 찾기
``` python
# 1 ~ N번 까지의 도시와 M개의 단방향 도로 존재
# X부터 출발하여 도달할 수 있는 도시 중에서 최단 거리가 K인 모든 도시의 번호를
# 한 줄에 하나씩 오름차순으로 정렬

# N, M, K, X = map(int, input().split())
N, M, K, X = 4, 4, 1, 1
# graph = [[]*(N+1)]
# for _ in range(M):
#     inn = list(map(int, input().split()))
#     graph[inn[0]].extend(inn[1])
# for row in graph:
#     row.sort()
graph = [[], [2, 3], [3, 4], [], []]

start = X
q = []
q.append(start)
visited = [0] * (N+1)
visited[start] = 1
answer = []
while q:
    i = q.pop(0)
    if visited[i] == K+1:
        answer.append(i)
        continue

    for data in graph[i]:

        if visited[data] == 0:
            ii = data
            visited[ii] =visited[i] + 1
            q.append(ii)
if not answer:
    print(-1)

answer.sort()
for i in range(len(answer)):
    print(answer[i])
```


## 연구소
``` python
# N x M 연구소의 크기
# 벽을 꼭 3개 세워야 함
# 0: 빈칸, 1: 벽, 2: 바이러스
# 바이러스는 상하좌우로 퍼져나감
# 벽이 세워졌을 때, 얻을 수 있는 안전 영역의 최댓값 구하기
N, M = map(int, input().split())
matrix = [list(map(int, input().split())) for _ in range(N)]

# 아이디어: 벽을 3개 설치할 수 있는 모든 경우의 수 계산
# 벽을 3개 설치했을 때 가지는 안전영역 구하기 -> 최솟값만

# 벽을 설치한 뒤의 맵 리스트
temp = [[0]*M for _ in range(N)]
result = 0
# 바이러스 계속 퍼짐
def virus(x, y):
    for dx, dy in [[1, 0], [-1, 0], [0, 1], [0, -1]]:
        nx, ny = x+dx, y+dy
        if 0<=nx<N and 0<=ny<M:
            if temp[nx][ny] == 0:
                temp[nx][ny] = 2
                virus(nx, ny)

# 안전 영역의 크기
def get_score():
    score = 0
    for i in range(N):
        for j in range(M):
            if temp[i][j] == 0 :
                score += 1
    return score 

# 깊이우선 탐색을 이용해 울타리를 설치하면서 매번 안전영역의 크기 계산
def dfs(count):
    global result
    # 울타리가 3개 설치된 경우
    if count == 3:
        for i in range(N):
            for j in range(M):
                temp[i][j] = matrix[i][j]
        for i in range(N):
            for j in range(M):
                if temp[i][j] == 2:
                    virus(i,j)
        result = max(result, get_score())
        return
    # 빈 곳에 울타리 설치
    for i in range(N):
        for j in range(M):
            if matrix[i][j] ==0:
                matrix[i][j] = 1
                count += 1
                # 백트레킹!!!
                dfs(count)
                matrix[i][j] = 0
                count-=1

dfs(0)
print(result)
```

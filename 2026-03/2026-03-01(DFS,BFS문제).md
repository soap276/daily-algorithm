## 특정 거리의 도시 찾기
``` python
# 1 ~ N번 까지의 도시와 M개의 단방향 도로 존재
# X부터 출발하여 도달할 수 있는 도시 중에서 최단 거리가 K인 모든 도시의 번호를
# 한 줄에 하나씩 오름차순으로 정렬

# N, M, K, X = map(int, input().split())
N, M, K, X = 4, 4, 1, 1
# graph = [[]*(N+1)]
# for _ in range(M):
#     inn = list(map(int, input().split()))
#     graph[inn[0]].extend(inn[1])
# for row in graph:
#     row.sort()
graph = [[], [2, 3], [3, 4], [], []]

start = X
q = []
q.append(start)
visited = [0] * (N+1)
visited[start] = 1
answer = []
while q:
    i = q.pop(0)
    if visited[i] == K+1:
        answer.append(i)
        continue

    for data in graph[i]:

        if visited[data] == 0:
            ii = data
            visited[ii] =visited[i] + 1
            q.append(ii)
if not answer:
    print(-1)

answer.sort()
for i in range(len(answer)):
    print(answer[i])
```


## 연구소
``` python
# N x M 연구소의 크기
# 벽을 꼭 3개 세워야 함
# 0: 빈칸, 1: 벽, 2: 바이러스
# 바이러스는 상하좌우로 퍼져나감
# 벽이 세워졌을 때, 얻을 수 있는 안전 영역의 최댓값 구하기
N, M = map(int, input().split())
matrix = [list(map(int, input().split())) for _ in range(N)]

# 아이디어: 벽을 3개 설치할 수 있는 모든 경우의 수 계산
# 벽을 3개 설치했을 때 가지는 안전영역 구하기 -> 최솟값만

# 벽을 설치한 뒤의 맵 리스트
temp = [[0]*M for _ in range(N)]
result = 0
# 바이러스 계속 퍼짐
def virus(x, y):
    for dx, dy in [[1, 0], [-1, 0], [0, 1], [0, -1]]:
        nx, ny = x+dx, y+dy
        if 0<=nx<N and 0<=ny<M:
            if temp[nx][ny] == 0:
                temp[nx][ny] = 2
                virus(nx, ny)

# 안전 영역의 크기
def get_score():
    score = 0
    for i in range(N):
        for j in range(M):
            if temp[i][j] == 0 :
                score += 1
    return score 

# 깊이우선 탐색을 이용해 울타리를 설치하면서 매번 안전영역의 크기 계산
def dfs(count):
    global result
    # 울타리가 3개 설치된 경우
    if count == 3:
        for i in range(N):
            for j in range(M):
                temp[i][j] = matrix[i][j]
        for i in range(N):
            for j in range(M):
                if temp[i][j] == 2:
                    virus(i,j)
        result = max(result, get_score())
        return
    # 빈 곳에 울타리 설치
    for i in range(N):
        for j in range(M):
            if matrix[i][j] ==0:
                matrix[i][j] = 1
                count += 1
                # 백트레킹!!!
                dfs(count)
                matrix[i][j] = 0
                count-=1

dfs(0)
print(result)
```
## 연구소
``` python
# 바이러스 종류는 1~K번, 
# 바이러스는 1초마다 상하좌우로 증식, 매초 번호가 낮은 종류의 바이러스부터 먼저 증식
# 증식 과정에서 특정 칸에 이미 어떤 바이러스가 들어가있다면 그곳에는 다른 바이러스 못 들어감

N,  K = map(int, input().split())
matrix = [list(map(int, input().split())) for _ in range(N)]
S, X, Y = map(int, input().split())
# N, K = 3, 3
# matrix = [[1, 0, 2], [0, 0, 0], [3, 0, 0]]
# S, X, Y = 2, 3, 2

visited = [[0] * N for _ in range(N)]

def virus(matrix):
    # 1. 이번 1초에 퍼질 감염원만 수집 (현재 시점의 바이러스들)
    sources = []
    for i in range(N):
        for j in range(N):
            if matrix[i][j] != 0:
                sources.append((matrix[i][j], i, j)) # (바이러스 번호, 좌표)
    
    # 번호 낮은 바이러스부터 퍼지게 정렬
    sources.sort()
    # 이번 1초 동안의 감염 결과만 기록할 visited (매초 초기화 !!!!!)
    visited = [[0]*N for _ in range(N)]

    # sources 에 있는 것들만 확산 (새로 감염된 칸은 이번 초에 다시 퍼지지 않음)
    for k, i, j in sources:
        for di, dj in [[1, 0], [-1, 0], [0, 1], [0, -1]]:
            ni, nj = i+di, j+dj
            if 0<=ni<N and 0<=nj<N:
                if matrix[ni][nj] == 0 and visited[ni][nj] == 0:
                    matrix[ni][nj] = k
                    visited[ni][nj] = 1
    return matrix

for _ in range(S):
    virus(matrix)

print(matrix[X-1][Y-1])
```

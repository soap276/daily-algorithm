### 리스트 거꾸로 뒤집어 반환 (list.reverse 내장함수 사용 금지)
``` python 
def reverse_list(visited_items):
    # visited_items.reverse() # 리스트의 순서를 거꾸로 뒤집어 정렬
    # return visited_items

    for i in range(len(visited_items) // 2):
        temp = visited_items[i]
        visited_items[i] = visited_items[(len(visited_items)-1) - i]
        visited_items[(len(visited_items)-1) - i] = temp
    return visited_items

print(reverse_list([1, 2, 3, 4, 5]))  # [5, 4, 3, 2, 1]
print(reverse_list(['A', 'B', 'C']))  # ['C', 'B', 'A']
```

### 딕셔너리 만들 때 value값 리스트로 한 개 이상일 때 
``` python 
def categorize_books(books):
    dic = {}
    for data in books:
        if data['category'] in dic:
            dic[data['category']].append(data['title']) ## 이 부분 기억하기!!!
        else:
            dic[data['category']] = [data['title']]
    return dic

books_data = [
    {'title': 'Python Basic', 'category': 'IT'},
    {'title': 'Java Standard', 'category': 'IT'},
    {'title': 'History of World', 'category': 'History'},
    {'title': 'Cooking 101', 'category': 'Life'}
]
# 결과:
# {
#   'IT': ['Python Basic', 'Java Standard'],
#   'History': ['History of World'],
#   'Life': ['Cooking 101']
# }
print(categorize_books(books_data))
```

### 유료 주차장 정산 시스템
```python
def calculate_parking_fee(parking_logs):
    sum_data = 0
    fee = 0
    for data in parking_logs:
        if data == 0: # 0분 주차 0원
            fee = 0
        elif data <= 30: # 30분 이하 2000원 (기본요금)
            fee = 2000
        elif data > 30: # 30분 초과 시 기본요금에 10분 당 500원 추가
            fee = 2000 + (((data-30) // 10) * 500) 
            if (data-30) % 10 > 0: # 31분도 10분 추가 요금 부과: 2500원
                fee += 500
        if fee > 10000: # 계산 요금이 10000원을 초과하면 10000원만 청구
            fee = 10000
        sum_data += fee
    return sum_data  

print(calculate_parking_fee([25, 45, 300])) # 15000
print(calculate_parking_fee([10, 0])) # 2000
```

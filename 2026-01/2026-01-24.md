## def 함수 활용
``` python
pro_num = 1000
global_data = {'subject': 'python', 'day': 3, 'title': '함수 활용하기'}

def create_data(subject, day, title = None):
    data = {}
    data['과목'] = subject
    data['일차'] = day
    data['제목'] = title
    global pro_num
    data['문제 번호'] = pro_num + 1 
    pro_num += 1
    return data

result1 = create_data('python', 3)
result2 = create_data('web', 1, 'web 연습하기')
result3 = create_data(*global_data) # 언패킹: 여러개의 값이 들어있는 변수를 하나씩 전달하는 것 (*: 시퀀스 언패킹, **:딕셔너리 언패킹)

print(result1)
print(result2)
print(result3)
```

## 재귀함수
``` python
def power(base, exponent):
    if exponent == 0:
        return 1
    else:
        return base * power(base, exponent-1)
    
def sum_of_digits(num):
    if num < 10:
        return num
    else: 
        return num%10 + sum_of_digits(num //10)
    
print(power(2, 3))
print(sum_of_digits(321))
```
## 도서 대여 서비스
``` python 
number_of_book = 100
number_of_people = 0

def decrease_book(borrow_book):
    global number_of_book
    number_of_book -= borrow_book
    print (f'남은 책의 수 : {number_of_book}')

def rental_book(info): # 단 하나의 인자를 받음. 인자는 {'고객이름': 고객나이} 형태의 딕셔너리
    name = list(info.keys())[0]  # name = list(info)[0]
    age = list(info.values())[0]  # age = info[name]
    book = age // 10
    decrease_book(book)
    print(f'{name}님이 {book}의 책을 대여하였습니다.')

# rental_book('홍길동', 3)

def increase_user():
    global number_of_people
    number_of_people += 1

def create_user(name, age, address):
    increase_user()
    user_info = {
        'name': name,
        'age': age,
        'address': address
    }
    print(f'{name}님 환영합니다!')
    return user_info

# ------ 실행 부분 (데이터 변환)⭐ ------
```
